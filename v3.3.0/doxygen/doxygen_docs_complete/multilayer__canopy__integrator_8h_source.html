166void multilayer_canopy_integrator::run() const
167{
168 double const LAIc = lai / nlayers;
169
170 double canopy_assimilation_rate{0.0};
171 double canopy_conductance{0.0};
172 double canopy_gross_assimilation_rate{0.0};
173 double canopy_photorespiration_rate{0.0};
174 double canopy_RL{0.0};
175 double canopy_transpiration_rate{0.0};
176
177 // Integrate assimilation, transpiration, and conductance throughout the
178 // canopy
179 for (int i = 0; i < nlayers; ++i) {
180 double const sunlit_lai = *sunlit_fraction_ips[i] * LAIc;
181 double const shaded_lai = *shaded_fraction_ips[i] * LAIc;
182
183 canopy_assimilation_rate += *sunlit_Assim_ips[i] * sunlit_lai +
184 *shaded_Assim_ips[i] * shaded_lai;
185
186 canopy_transpiration_rate += *sunlit_TransR_ips[i] * sunlit_lai +
187 *shaded_TransR_ips[i] * shaded_lai;
188
189 canopy_conductance += *sunlit_Gs_ips[i] * sunlit_lai +
190 *shaded_Gs_ips[i] * shaded_lai;
191
192 canopy_gross_assimilation_rate += *sunlit_GrossAssim_ips[i] * sunlit_lai +
193 *shaded_GrossAssim_ips[i] * shaded_lai;
194
195 canopy_photorespiration_rate += *sunlit_Rp_ips[i] * sunlit_lai +
196 *shaded_Rp_ips[i] * shaded_lai;
197
198 canopy_RL += *sunlit_RL_ips[i] * sunlit_lai +
199 *shaded_RL_ips[i] * shaded_lai;
200 }
201
202 // Calculate the rate of whole-plant growth respiration
203 double const whole_plant_gr =
204 growth_resp(canopy_assimilation_rate, growth_respiration_fraction); // micromol / m^2 / s
205
206 // For transpiration, we need to convert mmol / m^2 / s into Mg / ha / hr
207 // using the molar mass of water in kg / mol, which can be accomplished by
208 // the following conversion factor:
209 // (3600 s / hr) * (1e-3 mol / mmol) * (1e-3 Mg / kg) * (1e4 m^2 / ha)
210 // = 36 s * mol * Mg * m^2 / (hr * mmol * kg * ha)
211 const double cf2 = physical_constants::molar_mass_of_water * 36; // (Mg / ha / hr) / (mmol / m^2 / s)
212
213 update(canopy_assimilation_molar_flux_op, canopy_assimilation_rate - whole_plant_gr);
214 update(canopy_conductance_op, canopy_conductance);
215 update(canopy_gross_assimilation_molar_flux_op, canopy_gross_assimilation_rate);
216 update(canopy_photorespiration_molar_flux_op, canopy_photorespiration_rate);
217 update(canopy_RL_molar_flux_op, canopy_RL);
218 update(canopy_transpiration_rate_op, canopy_transpiration_rate * cf2);
219 update(whole_plant_growth_respiration_molar_flux_op, whole_plant_gr);
220}
