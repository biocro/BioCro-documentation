109void maintenance_respiration::do_operation() const
110{
111 double Tref = 25.0; // reference temperature for the Q10 function
112
113 double dLeaf = -Leaf * mrc_leaf * Q10_temperature_response(temp, Tref); // Mg / ha
114
115 double dStem = -Stem * mrc_stem * Q10_temperature_response(temp, Tref);
116
117 double dRoot = -Root * mrc_root * Q10_temperature_response(temp, Tref);
118 //assume rhizome has the same maintenance_respiration_coef as root
119 double dRhizome = -Rhizome * mrc_root * Q10_temperature_response(temp, Tref);
120
121 double dGrain = -Grain * mrc_grain * Q10_temperature_response(temp, Tref);
122 //assume shell has the same maintenance_respiration_coef as grain
123 double dShell = -Shell * mrc_grain * Q10_temperature_response(temp, Tref); // Mg / ha
124
125 update(Leaf_op, dLeaf); // Mg / ha
126 update(Stem_op, dStem); // Mg / ha
127 update(Root_op, dRoot); // Mg / ha
128 update(Rhizome_op, dRhizome); // Mg / ha
129 update(Grain_op, dGrain); // Mg / ha
