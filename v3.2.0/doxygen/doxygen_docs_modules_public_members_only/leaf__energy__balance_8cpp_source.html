31double vapor_density_from_pressure(
32 double density_of_dry_air, // kg / m^3
33 double total_pressure, // Pa
34 double vapor_pressure // Pa
35)
36{
37 // Specify the ratio of the relative molecular mass of water to the relative
38 // molecular mass of dry air. See Thornley & Johnson (1990), page 409.
39 constexpr double molecular_ratio_water_air = 0.622; // dimensionless
40
41 return density_of_dry_air * molecular_ratio_water_air *
42 vapor_pressure / (total_pressure - vapor_pressure); // kg / m^3
43}

91 double wind_speed // m / s
92)

94 // Set some constants
95 double constexpr epsilon_s = 1.0; // dimensionless
96
97 // Get water vapor and air properties based on the air temperature
98 const double c_p = TempToCp(air_temperature); // J / kg / K
99 const double lambda = water_latent_heat_of_vaporization_henderson(air_temperature); // J / kg
100 const double p_w_sat_air = saturation_vapor_pressure(air_temperature); // Pa
101 const double rho_ta = dry_air_density(air_temperature, air_pressure); // kg / m^3
102 const double s = TempToSFS(air_temperature); // kg / m^3 / K
103
104 // Get the pyschrometric parameter
105 const double gamma = rho_ta * c_p / lambda; // kg / m^3 / K
106
107 // Get vapor density in the ambient air.
108 const double p_w_air = p_w_sat_air * relative_humidity; // Pa
109
110 const double rho_w_air =
111 vapor_density_from_pressure(rho_ta, air_pressure, p_w_air); // kg / m^3
112
113 // Get vapor density deficit
114 const double rho_w_sat =
115 vapor_density_from_pressure(rho_ta, air_pressure, p_w_sat_air); // kg / m^3
116
117 const double Delta_rho = rho_w_sat - rho_w_air; // kg / m^3
118
119 // Get total absorbed light energy (longwave and shortwave)
120 const double J_a = absorbed_shortwave_energy + absorbed_longwave_energy; // J / m^2 / s
121
122 // Make an initial guess for the leaf temperature
123 double leaf_temperature{air_temperature + 0.1}; // degrees C
124
125 // Initialize other loop variables; their values will be set during the loop
126 double gbw_leaf{}; // m / s
127 double gbw{}; // m / s
128 double gsw{}; // m / s
129 double gw{}; // m / s
130 double mv_tl{}; // m^3 / mol
131 double Phi_N{}; // J / m^2 / s
132 double pm_bottom{}; // J / m^3 / K
133 double pm_top{}; // J / m^3
134 double R_l{}; // J / m^2 / s
135
136 // Run loop to find the leaf temperature
137 double constexpr max_delta{10}; // degrees C
138 double old_leaf_temperature{}; // degrees C
139 double change_in_tl{}; // degrees C
140 int counter{0};
141
142 do {
143 // Store leaf temperature from previous loop iteration
144 old_leaf_temperature = leaf_temperature; // degrees C
145
146 // Get longwave energy losses
147 R_l = epsilon_s * physical_constants::stefan_boltzmann *
148 pow(conversion_constants::celsius_to_kelvin + leaf_temperature, 4); // J / m^2 / s
149
150 // Get the energy available for transpiration and heat loss
151 Phi_N = J_a - R_l; // J / m^2 / s
152
153 // Get stomatal conductance to water vapor as a mass conductance
154 mv_tl = molar_volume(leaf_temperature, air_pressure); // m^3 / mol
155 gsw = stomatal_conductance * mv_tl; // m / s
156
157 // Get leaf boundary layer conductance to water vapor
158 gbw_leaf = leaf_boundary_layer_conductance_nikolov(
159 air_temperature,
160 leaf_temperature - air_temperature,
161 p_w_air,
162 gsw,
163 leaf_width,
164 wind_speed,
165 air_pressure); // m / s
166
167 // Get the boundary layer conductance and total conductance to water
168 // vapor
169 gbw = 1.0 / (1.0 / gbw_leaf + 1.0 / gbw_canopy); // m / s
170 gw = 1.0 / (1.0 / gsw + 1.0 / gbw); // m / s
171
172 // Get the new leaf temperature using the Penman-Monteith equation
173 pm_top = Phi_N / gw - lambda * Delta_rho; // J / m^3
174 pm_bottom = lambda * (s + gamma * (1.0 + gbw / gsw)); // J / m^3 / K
175
176 leaf_temperature = air_temperature +
177 std::min(std::max(pm_top / pm_bottom, -max_delta), max_delta); // degrees C
178
179 // Get the change in leaf temperature relative to the previous iteration
180 change_in_tl = std::abs(leaf_temperature - old_leaf_temperature); // degrees C
181
182 } while ((++counter <= 50) && (change_in_tl > 0.25));
183
184 // Calculate additional outputs
185 double const Delta_T = leaf_temperature - air_temperature; // degrees C
186 double const E = (Delta_rho + s * Delta_T) * gw; // kg / m^2 / s
187 double const H = rho_ta * c_p * Delta_T * gbw; // J / m^2 / s
188 double const storage = Phi_N - H - lambda * E; // J / m^2 / s
189
190 // Relative humidity just outside the leaf boundary layer
191 double const RH_canopy = (rho_w_air - E / gbw_canopy) / rho_w_sat; // dimensionless
192
193 // Potential evapotranspiration can be calculated assuming infinite stomatal
194 // conductance; here we call this "Penman transpiration."
195 double const EPen = (s * Phi_N + lambda * gamma * gbw * Delta_rho) /
196 (lambda * (s + gamma)); // kg / m^2 / s
197
198 // Evapotranspiration according to Priestly (?)
199 double constexpr dryness_coefficient = 1.26; // dimensionless
200 double const EPries = dryness_coefficient * s * Phi_N /
201 (lambda * (s + gamma)); // kg / m^2 / s
202
203 // The transpiration rates here have units of kg / m^2 / s. They can be
204 // converted to mmol / m^2 / s using the molar mass of water (in kg / mol)
205 // and noting that 1e3 mmol = 1 mol.
206 double constexpr cf = 1e3 / physical_constants::molar_mass_of_water; // mmol / kg for water
207
208 return energy_balance_outputs{
209 /* Deltat = */ Delta_T, // degrees C
210 /* E_loss = */ lambda * E, // J / m^2 / s
211 /* EPenman = */ EPen * cf, // mmol / m^2 / s
212 /* EPriestly = */ EPries * cf, // mmol / m^2 / s
213 /* gbw = */ gbw, // m / s
214 /* gbw_canopy = */ gbw_canopy, // m / s
215 /* gbw_leaf = */ gbw_leaf, // m / s
216 /* gbw_molecular = */ gbw / mv_tl, // mol / m^2 / s
217 /* gsw = */ gsw, // m / s
218 /* H = */ H, // J / m^2 / s
219 /* PhiN = */ Phi_N, // J / m^2 / s
220 /* RH_canopy = */ RH_canopy, // dimensionless
221 /* storage = */ storage, // J / m^2 / s
222 /* TransR = */ E * cf, // mmol / m^2 / s
223 /* iterations = */ counter // not a physical quantity
224 };
