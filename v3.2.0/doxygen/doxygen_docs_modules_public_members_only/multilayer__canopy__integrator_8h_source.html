153void multilayer_canopy_integrator::run() const
154{
155 double const LAIc = lai / nlayers;
156 double canopy_assimilation_rate{0.0};
157 double canopy_transpiration_rate{0.0};
158 double canopy_conductance{0.0};
159 double GrossAssim{0.0};
160 double canopy_photorespiration_rate{0.0};
161
162 // Integrate assimilation, transpiration, and conductance throughout the
163 // canopy
164 for (int i = 0; i < nlayers; ++i) {
165 double const sunlit_lai = *sunlit_fraction_ips[i] * LAIc;
166 double const shaded_lai = *shaded_fraction_ips[i] * LAIc;
167
168 canopy_assimilation_rate += *sunlit_Assim_ips[i] * sunlit_lai +
169 *shaded_Assim_ips[i] * shaded_lai;
170
171 canopy_transpiration_rate += *sunlit_TransR_ips[i] * sunlit_lai +
172 *shaded_TransR_ips[i] * shaded_lai;
173
174 canopy_conductance += *sunlit_Gs_ips[i] * sunlit_lai +
175 *shaded_Gs_ips[i] * shaded_lai;
176
177 GrossAssim += *sunlit_GrossAssim_ips[i] * sunlit_lai +
178 *shaded_GrossAssim_ips[i] * shaded_lai;
179
180 canopy_photorespiration_rate += *sunlit_Rp_ips[i] * sunlit_lai +
181 *shaded_Rp_ips[i] * shaded_lai;
182 }
183
184 // Modify net assimilation to account for respiration
185 // Note: this was originally only done for the C3 canopy
186 // Note: it seems like this should not be necessary since the assimilation
187 // model includes respiration
188 double const growth_respiration =
189 growth_respiration_fraction * fabs(canopy_assimilation_rate); // micromol / m^2 / s
190
191 canopy_assimilation_rate =
192 canopy_assimilation_rate - growth_respiration; // micromol / m^2 / s
193
194 // For transpiration, we need to convert mmol / m^2 / s into Mg / ha / hr
195 // using the molar mass of water in kg / mol, which can be accomplished by
196 // the following conversion factor:
197 // (3600 s / hr) * (1e-3 mol / mmol) * (1e-3 Mg / kg) * (1e4 m^2 / ha)
198 // = 36 s * mol * Mg * m^2 / (hr * mmol * kg * ha)
199 const double cf2 = physical_constants::molar_mass_of_water * 36; // (Mg / ha / hr) / (mmol / m^2 / s)
200
201 update(canopy_assimilation_rate_CO2_op, canopy_assimilation_rate);
202
203 update(GrossAssim_CO2_op, GrossAssim);
204
205 update(canopy_transpiration_rate_op, canopy_transpiration_rate * cf2);
206
207 update(canopy_conductance_op, canopy_conductance);
208
209 update(canopy_photorespiration_rate_CO2_op, canopy_photorespiration_rate);
210}
