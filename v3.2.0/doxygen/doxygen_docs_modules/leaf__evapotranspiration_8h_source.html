121void leaf_evapotranspiration::do_operation() const
122{
123 // Get absorbed longwave radiation
124 double const absorbed_longwave =
125 1.0 * physical_constants::stefan_boltzmann *
126 pow(conversion_constants::celsius_to_kelvin + temp, 4); // J / m^2 / s
127
128 // Get canopy boundary layer conductance to water vapor
129 double const gbw_canopy = canopy_boundary_layer_conductance_thornley(
130 canopy_height,
131 windspeed,
132 min_gbw_canopy,
133 wind_speed_height); // m / s
134
135 energy_balance_outputs result = leaf_energy_balance(
136 absorbed_longwave,
137 absorbed_shortwave,
138 atmospheric_pressure,
139 temp,
140 gbw_canopy,
141 leafwidth,
142 rh,
143 Gs,
144 windspeed);
145
146 update(EPenman_op, result.EPenman);
147 update(EPriestly_op, result.EPriestly);
148 update(E_loss_op, result.E_loss);
149 update(gbw_canopy_op, result.gbw_canopy);
150 update(gbw_leaf_op, result.gbw_leaf);
151 update(gbw_op, result.gbw);
152 update(gsw_op, result.gsw);
153 update(H_op, result.H);
154 update(iterations_op, result.iterations);
155 update(leaf_temperature_op, temp + result.Deltat);
156 update(PhiN_op, result.PhiN);
157 update(storage_op, result.storage);
158 update(TransR_op, result.TransR);
159}
