203void partitioning_growth_calculator_leaf_costs::do_operation() const
204{
205 // Specify the base temperature for Q10 growth respiration
206 double constexpr Tref = 0.0; // degrees C
207
208 // Calculate the base rate of new leaf production, accounting for water
209 // stress and the associated respiratory costs (Mg / ha / hr)
210 double const base_rate_leaf{
211 kLeaf <= 0 ? 0
212 : canopy_assim < 0 ? canopy_assim
213 : canopy_assim * kLeaf};
214
215 double const Leaf_WS_loss_rate{growth_resp(
216 base_rate_leaf,
217 (1.0 - LeafWS))};
218
219 double const Leaf_gr_rate{growth_resp_Q10(
220 base_rate_leaf - Leaf_WS_loss_rate, grc_leaf, temp, Tref)};
221
222 // Determine the carbon flux to use for the non-leaf organs (Mg / ha / hr)
223 double const nonleaf_carbon_flux{canopy_assim > 0 ? canopy_assim : 0};
224
225 // Calculate the base rate of new stem production and the associated
226 // respiratory costs (Mg / ha / hr)
227 double const base_rate_stem{kStem > 0 ? nonleaf_carbon_flux * kStem : 0};
228 double const Stem_gr_rate{growth_resp_Q10(base_rate_stem, grc_stem, temp, Tref)};
229
230 // Calculate the base rate of new root production and the associated
231 // respiratory costs (Mg / ha / hr)
232 double const base_rate_root{kRoot > 0 ? nonleaf_carbon_flux * kRoot : 0};
233 double const Root_gr_rate{growth_resp_Q10(base_rate_root, grc_root, temp, Tref)};
234
235 // Calculate the base rate of new rhizome production and the associated
236 // respiratory costs (Mg / ha / hr)
237 double const base_rate_rhizome{kRhizome > 0 ? nonleaf_carbon_flux * kRhizome : 0};
238 double const Rhizome_gr_rate{growth_resp_Q10(base_rate_rhizome, grc_rhizome, temp, Tref)};
239
240 // Calculate the base rate of new grain production (Mg / ha / hr)
241 double const base_rate_grain{kGrain > 0 ? nonleaf_carbon_flux * kGrain : 0};
242 double const Grain_gr_rate{growth_resp_Q10(base_rate_grain, grc_grain, temp, Tref)};
243
244 // Calculate the base rate of new shell production (Mg / ha / hr)
245 double const base_rate_shell{kShell > 0 ? nonleaf_carbon_flux * kShell : 0};
246 double const Shell_gr_rate{growth_resp_Q10(base_rate_shell, grc_shell, temp, Tref)};
247
248 // Update the output quantity list
249 update(Grain_gr_rate_op, Grain_gr_rate);
250 update(Leaf_gr_rate_op, Leaf_gr_rate);
251 update(Leaf_WS_loss_rate_op, Leaf_WS_loss_rate);
252 update(net_assimilation_rate_grain_op, base_rate_grain - Grain_gr_rate);
253 update(net_assimilation_rate_leaf_op, base_rate_leaf - Leaf_gr_rate - Leaf_WS_loss_rate);
254 update(net_assimilation_rate_rhizome_op, base_rate_rhizome - Rhizome_gr_rate);
255 update(net_assimilation_rate_root_op, base_rate_root - Root_gr_rate);
256 update(net_assimilation_rate_shell_op, base_rate_shell - Shell_gr_rate);
257 update(net_assimilation_rate_stem_op, base_rate_stem - Stem_gr_rate);
258 update(Rhizome_gr_rate_op, Rhizome_gr_rate);
259 update(Root_gr_rate_op, Root_gr_rate);
260 update(Shell_gr_rate_op, Shell_gr_rate);
261 update(Stem_gr_rate_op, Stem_gr_rate);
262}
