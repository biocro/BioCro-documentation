202void partitioning_growth_calculator::do_operation() const
203{
204 // Specify the base temperature for Q10 growth respiration
205 double constexpr Tref = 0.0; // degrees C
206
207 // Calculate the base rate of new leaf production, accounting for water
208 // stress and the associated respiratory costs (Mg / ha / hr)
209 double const base_rate_leaf{kLeaf > 0 ? canopy_assim * kLeaf : 0};
210
211 double const Leaf_WS_loss_rate{growth_resp(
212 base_rate_leaf,
213 (1.0 - LeafWS))};
214
215 double const Leaf_gr_rate{growth_resp_Q10(
216 base_rate_leaf - Leaf_WS_loss_rate, grc_leaf, temp, Tref)};
217
218 // Calculate the base rate of new stem production and the associated
219 // respiratory costs (Mg / ha / hr)
220 double const base_rate_stem{kStem > 0 ? canopy_assim * kStem : 0};
221 double const Stem_gr_rate{growth_resp_Q10(base_rate_stem, grc_stem, temp, Tref)};
222
223 // Calculate the base rate of new root production and the associated
224 // respiratory costs (Mg / ha / hr)
225 double const base_rate_root{kRoot > 0 ? canopy_assim * kRoot : 0};
226 double const Root_gr_rate{growth_resp_Q10(base_rate_root, grc_root, temp, Tref)};
227
228 // Calculate the base rate of new rhizome production and the associated
229 // respiratory costs (Mg / ha / hr)
230 double const base_rate_rhizome{kRhizome > 0 ? canopy_assim * kRhizome : 0};
231 double const Rhizome_gr_rate{growth_resp_Q10(base_rate_rhizome, grc_rhizome, temp, Tref)};
232
233 // Calculate the base rate of new grain production (Mg / ha / hr)
234 double const base_rate_grain{kGrain > 0 ? canopy_assim * kGrain : 0};
235 double const Grain_gr_rate{growth_resp_Q10(base_rate_grain, grc_grain, temp, Tref)};
236
237 // Calculate the base rate of new shell production (Mg / ha / hr)
238 double const base_rate_shell{kShell > 0 ? canopy_assim * kShell : 0};
239 double const Shell_gr_rate{growth_resp_Q10(base_rate_shell, grc_shell, temp, Tref)};
240
241 // Update the output quantity list
242 update(Grain_gr_rate_op, Grain_gr_rate);
243 update(Leaf_gr_rate_op, Leaf_gr_rate);
244 update(Leaf_WS_loss_rate_op, Leaf_WS_loss_rate);
245 update(net_assimilation_rate_grain_op, base_rate_grain - Grain_gr_rate);
246 update(net_assimilation_rate_leaf_op, base_rate_leaf - Leaf_gr_rate - Leaf_WS_loss_rate);
247 update(net_assimilation_rate_rhizome_op, base_rate_rhizome - Rhizome_gr_rate);
248 update(net_assimilation_rate_root_op, base_rate_root - Root_gr_rate);
249 update(net_assimilation_rate_shell_op, base_rate_shell - Shell_gr_rate);
250 update(net_assimilation_rate_stem_op, base_rate_stem - Stem_gr_rate);
251 update(Rhizome_gr_rate_op, Rhizome_gr_rate);
252 update(Root_gr_rate_op, Root_gr_rate);
253 update(Shell_gr_rate_op, Shell_gr_rate);
254 update(Stem_gr_rate_op, Stem_gr_rate);
255}
